o ".csv" do nome do arquivo df['ano'] = ano =int(ano.replace('.csv','')). Append para add a coluna na base'''

for arquivo in caminho_bases.iterdir():
    ano = arquivo.name[-8:]
    ano =int(ano.replace('.csv',''))
    df = pd.read_csv(caminho_bases / arquivo.name)
    df['ano'] = ano
    base = base.append(df)
